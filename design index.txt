    <script>
        // Tooltip initialization
        const tooltipTriggerList = [].slice.call(document.querySelectorAll('[data-bs-toggle="tooltip"]'))
        const tooltipList = tooltipTriggerList.map(function (tooltipTriggerEl) {
            return new bootstrap.Tooltip(tooltipTriggerEl)
        });

        // Quick View Modal
        const quickViewModal = document.getElementById('quickViewModal');
        quickViewModal.addEventListener('show.bs.modal', event => {
            const button = event.relatedTarget;
            const id = button.getAttribute('data-id');
            const name = button.getAttribute('data-name');
            const description = button.getAttribute('data-description');
            const price = button.getAttribute('data-price');
            const image = button.getAttribute('data-image');

            document.getElementById('quick-view-image').src = image;
            document.getElementById('quick-view-name').textContent = name;
            document.getElementById('quick-view-description').textContent = description;
            document.getElementById('quick-view-price').textContent = price;
            document.getElementById('quick-view-add-to-cart').setAttribute('data-id', id);
            document.getElementById('quick-view-add-to-cart').setAttribute('data-name', name);
            document.getElementById('quick-view-add-to-cart').setAttribute('data-price', price);
        });

        // Add to cart functionality
        document.querySelectorAll('.add-to-cart, #quick-view-add-to-cart').forEach(button => {
            button.addEventListener('click', function() {
                const productId = this.dataset.id;
                const productName = this.dataset.name;
                const productPrice = this.dataset.price;
                const quantity = 1; // Default quantity

                const formData = new FormData();
                formData.append('product_id', productId);
                formData.append('product_name', productName);
                formData.append('product_price', productPrice);
                formData.append('quantity', quantity);
                formData.append('total_price', productPrice * quantity);

                fetch('add_to_cart.php', {
                    method: 'POST',
                    body: formData
                })
                .then(response => response.json())
                .then(data => {
                    if (data.status === 'success') {
                        alert('Product added to cart!');
                        document.getElementById('cart-count').textContent = data.cart_count;
                    } else {
                        alert('Failed to add product to cart.');
                    }
                })
                .catch(error => console.error('Error:', error));
            });
        });
    </script>
