    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>
    <script>
    const productPrice = <?= json_encode($product['price']) ?>;

    document.getElementById('orderForm').addEventListener('submit', function(event) {
        event.preventDefault();
        const form = event.target;
        const formData = new FormData(form);

        fetch('process_order.php', {
            method: 'POST',
            body: formData
        }).then(response => response.text()).then(data => {
            if (data.trim() === 'success') {
                new bootstrap.Modal(document.getElementById('orderSuccessModal')).show();
                form.reset();
                document.getElementById('totalPrice').innerText = productPrice.toFixed(2);
                document.getElementById('quantity').value = 1;
            } else if (data.trim() === 'captcha_failed') {
                new bootstrap.Modal(document.getElementById('captchaFailureModal')).show();
            } else {
                alert('Error placing order.');
                console.log(data);
            }
        });
    });

    document.getElementById('increaseQuantity').addEventListener('click', function() {
        const quantityInput = document.getElementById('quantity');
        let quantity = parseInt(quantityInput.value);
        quantity++;
        quantityInput.value = quantity;
        updateTotalPrice();
    });

    document.getElementById('decreaseQuantity').addEventListener('click', function() {
        const quantityInput = document.getElementById('quantity');
        let quantity = parseInt(quantityInput.value);
        if (quantity > 1) {
            quantity--;
            quantityInput.value = quantity;
            updateTotalPrice();
        }
    });

    function updateTotalPrice() {
        const quantity = parseInt(document.getElementById('quantity').value);
        const totalPrice = quantity * productPrice;
        document.getElementById('totalPrice').innerText = totalPrice.toFixed(2);
        document.getElementById('totalPriceInput').value = totalPrice.toFixed(2);
        document.getElementById('quantityInput').value = quantity;
    }
    </script>
